## I. Question

Implement the `myAtoi(string s)` function, which converts a string to a 32-bit signed integer.

The algorithm for `myAtoi(string s)` is as follows:

1. **Whitespace**: Ignore any leading whitespace (`" "`).
2. **Signedness**: Determine the sign by checking if the next character is `'-'` or `'+'`, assuming positivity if neither present.
3. **Conversion**: Read the integer by skipping leading zeros until a non-digit character is encountered or the end of the string is reached. If no digits were read, then the result is 0.
4. **Rounding**: If the integer is out of the 32-bit signed integer range `[-231, 231 - 1]`, then round the integer to remain in the range. Specifically, integers less than `-231` should be rounded to `-231`, and integers greater than `231 - 1` should be rounded to `231 - 1`.

Return the integer as the final result.

- $$ 0 \leqq s.length \leqq 200 $$
- `s` consists of English letters (lower-case and upper-case), digits (`0-9`), `' '`, `'+'`, `'-'`, and `'.'`.

## Ⅱ. Solution

### A. Naive

- Time Complexity: $$ O(n) $$
- Space Complexity: $$ O(1) $$

#### 1. C++

- Runtime: 0 ms
- Memory: 9.28MB

```C++
class Solution {
public:
    int myAtoi(string s) {
        int n = s.length();
        int i = 0;

        // 1. Whitespace
        while (i < n && s[i] == ' ') {
            i++;
        }
        if (i == n) return 0;

        // 2. Signedness
        int sign = 1;
        if (s[i] == '+' || s[i] == '-') {
            sign = (s[i] == '-') ? -1 : 1;
            i++;
        }

        long ans = 0;
        
        // 3. Conversion & 4. Rounding
        while (i < n && isdigit(s[i])) {
            ans = ans * 10 + (s[i] - '0');
            
            // Optimization: Early exit on overflow
            if (sign * ans <= INT_MIN) return INT_MIN;
            if (sign * ans >= INT_MAX) return INT_MAX;
            
            i++;
        }

        return (int)(sign * ans);
    }
};
```
#### 2. Rust

- Runtime: 0 ms

```rust
impl Solution {
    pub fn my_atoi(s: String) -> i32 {
        // iter(): An iterator that retrieves array elements one by one by reference
        // peekable(): lets us look at(refer) the next character without consuming it (like a window).
        // from 'iter().peekable()', Double reference(&&) is needed
        let mut bytes = s.as_bytes().iter().peekable();

        // 1. WhiteSpace
        while let Some(&&b' ') = bytes.peek() { bytes.next();}

        // 2. Signedness
        let mut sign = 1i64;
        if let Some(&&ch) = bytes.peek() {
            if ch == b'+' || ch == b'-' {
                if ch == b'-' {sign = -1;}
                bytes.next();
            }
        }

        // 3. Conversion & 4. Rounding
        let mut num: i64 = 0;

        while let Some(&&ch) = bytes.peek() {
            if ch >= b'0' && ch <= b'9' {
                let digit = (ch - b'0') as i64;
                num = num * 10 + digit;

                let curr_val = sign * num;
                if curr_val >= i32::MAX as i64 {return i32::MAX}
                
                if curr_val <= i32::MIN as i64 {return i32::MIN}

                bytes.next();
            } else {
                break;
            }
        }

        (sign * num) as i32
    }
}
```

```go
func myAtoi(s string) int {
    n := len(s)
    i := 0
    
    for i < n && s[i] == ' ' {i++}

    if i == n {return 0}

    sign := 1
    if s[i] == '-' {
        sign = -1
        i++
    } else if s[i] == '+' {
        i++
    }

    var num int64 = 0

    for i < n && s[i] >= '0' && s[i] <= '9' {

        digit := int64(s[i] - '0')
        num = num * 10 + digit

        if sign * int(num) >= math.MaxInt32 {
            return math.MaxInt32
        }

        if sign * int(num) <= math.MinInt32 {
            return math.MinInt32
        }

        i++
    }

    return sign * int(num)
}
```

#### 4. Python

- Runtime: 1 ms
- Memory: 19.58 MB

```python
class Solution:
    def myAtoi(self, s: str) -> int:
        INT_MAX = 2**31 - 1
        INT_MIN = -2**31
        
        n = len(s)
        i = 0
        
        while i < n and s[i] == ' ':
            i += 1
            
        if i == n:
            return 0
            
        sign = 1
        if s[i] == '-':
            sign = -1
            i += 1
        elif s[i] == '+':
            i += 1
            
        num = 0
        while i < n and s[i].isdigit():
            num = num * 10 + (ord(s[i]) - ord('0'))
            
            if sign * num >= INT_MAX:
                return INT_MAX
            if sign * num <= INT_MIN:
                return INT_MIN
                
            i += 1
            
        return sign * num
```
