# 18. 4Sum

\#Medium #Two Pointer #DFS

[toc]

## 1. Question

Given an array `nums` of `n` integers, return *an array of all the **unique** quadruplets* `[nums[a], nums[b], nums[c], nums[d]]` such that:

- `0 <= a, b, c, d < n`
- `a`, `b`, `c`, and `d` are **distinct**.
- `nums[a] + nums[b] + nums[c] + nums[d] == target`

You may return the answer in **any order**.

**Example 1:**

> Input: nums = [1,0,-1,0,-2,2], target = 0
> Output: [[-2,-1,1,2],[-2,0,0,2],[-1,0,0,1]]

**Example 2:**

> Input: nums = [2,2,2,2,2], target = 8
> Output: [[2,2,2,2]]

**Constraints:**

- $$ 1 \leqq nums.length \leqq 200 $$
- $$ -10^9 \leqq nums[i] \leqq 10^9 $$
- $$ -10^9 \leqq target \leqq 10^9 $$

## 2. Solution

### 1) Two Pointer w/ Iterative

- Time Complexity: $$ O(n^3) $$
- Space Complexity: $$ O(1) $$

#### A. C++

- Runtime: 7 ms
- Memory: 17.43MB

```C++
class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        const int n = nums.size();
        if (n < 4) return {};

        vector<vector<int>> ans;

        ranges::sort(nums);
        
        // 'long long' guaranteed to be at least 64 bits.
        // 'long' is highly unpredictable across operating systems
        // long (Windows)	32-bit
        // long (Linux/macOS 64-bit)	64-bit
        // long long (All platforms)	64-bit
        for (int i = 0; i < n - 3; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            if (static_cast<long long>(nums[i]) + nums[i + 1] + nums[i + 2] + nums[i + 3] > target) break;
            if (static_cast<long long>(nums[i]) + nums[n - 1] + nums[n - 2] + nums[n - 3] < target) continue;

            for (int j = i + 1; j < n - 2; j++) {
                if (j > i + 1 && nums[j] == nums[j - 1]) continue;

                if (static_cast<long long>(nums[i]) + nums[j] + nums[j + 1] + nums[j + 2] > target) break;
                if (static_cast<long long>(nums[i]) + nums[j] + nums[n - 1] + nums[n - 2] < target) continue;

                int l = j + 1;
                int r = n - 1;

                while (l < r) {
                    long long sum = static_cast<long long>(nums[i]) + nums[j] + nums[l] + nums[r];

                    if (sum == target) {
                        ans.push_back({nums[i], nums[j], nums[l++], nums[r--]});
                        while (l < r && nums[l] == nums[l - 1]) l++;
                        while (l < r && nums[r] == nums[r + 1]) r--;
                    } else if (sum < target) {
                        l++;
                    } else {
                        r--;
                    };
                };
            };
        };
        return ans;
    }
};
```

### 2) Two Pointer w/ Recursive DFS

- Time Complexity: $$ O(n^3) $$
- Space Complexity: $$ O(1) $$

#### A. C++

- Runtime: 0 ms
- Memory: 17.44MB

```C++
class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        vector<vector<int>> ans;
        vector<int> path;
        ranges::sort(nums);
        // Cast target to long long to handle sums that exceed 32-bit int
        nSum(nums, 4, static_cast<long long>(target), 0, nums.size() - 1, path, ans);
        return ans;
    }

private:
    void nSum(const vector<int>& nums, const int n, const long long target, int l, int r, vector<int>& path, vector<vector<int>>& ans) {
        // Core boundary and pruning checks
        if (r - l + 1 < n) return;
        
        // Use 1LL to force 64-bit long long multiplication and prevent overflow
        // 1LL = long long 1
        if (1LL * nums[l] * n > target || 1LL * nums[r] * n < target) return;

        // Base case: 2Sum using the two-pointer approach
        if (n == 2) {
            while (l < r) {
                const long long sum = static_cast<long long>(nums[l]) + nums[r];
                if (sum == target) {
                    path.push_back(nums[l]);
                    path.push_back(nums[r]);
                    ans.push_back(path);
                    path.pop_back();
                    path.pop_back();
                    l++;
                    r--;
                    while (l < r && nums[l] == nums[l - 1]) l++;
                    while (l < r && nums[r] == nums[r + 1]) r--;
                } else if (sum < target) {
                    l++;
                } else {
                    r--;
                }
            }
            return;
        }

        // Recursive step for N > 2
        for (int i = l; i <= r; i++) {
            if (i > l && nums[i] == nums[i - 1]) continue; // Skip duplicates
            path.push_back(nums[i]);
            nSum(nums, n - 1, target - nums[i], i + 1, r, path, ans);
            path.pop_back(); // Backtrack
        }
    }
};
```

```Go
	var path []int
	
	slices.Sort(nums)
	
	// Convert target to int64 up front to handle potential 32-bit overflows
	nSum(nums, 4, int64(target), 0, len(nums)-1, &path, &ans)
	return ans

func nSum(nums []int, n int, target int64, l, r int, path *[]int, ans *[][]int) {
	// Core boundary check
	if r - l + 1 < n {return}

	// Pruning checks using int64 to avoid overflow
	if int64(nums[l])*int64(n) > target || int64(nums[r])*int64(n) < target {return}

	// Base case: 2Sum using the two-pointer approach
	if n == 2 {
		for l < r {
			sum := int64(nums[l]) + int64(nums[r])
			if sum == target {
				// Append current pair to the path
				*path = append(*path, nums[l], nums[r])
				
				// Idiomatic Go slice copy: We MUST copy the slice before adding to 'ans'.
				// Since slices are reference types, appending 'path' directly would cause 
				// downstream mutations to overwrite previous results.
				temp := make([]int, len(*path))
				copy(temp, *path)
				*ans = append(*ans, temp)
				
				// Backtrack: Remove the 2 elements we just added
				*path = (*path)[:len(*path)-2]
				
				l++
				r--
				
				// Skip duplicates
				for l < r && nums[l] == nums[l-1] {
					l++
				}
				for l < r && nums[r] == nums[r+1] {
					r--
				}
			} else if sum < target {
				l++
			} else {
				r--
			}
		}
		return
	}

	// Recursive step for N > 2
	for i := l; i <= r; i++ {
		if i > l && nums[i] == nums[i-1] {
			continue // Skip duplicates
		}
		
		// Choose
		*path = append(*path, nums[i])
		
		// Explore
		nSum(nums, n-1, target-int64(nums[i]), i+1, r, path, ans)
		
		// Unchoose (Backtrack)
		*path = (*path)[:len(*path)-1]
	}
}
```

```python
class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        ans = []

        nums.sort()

        def nSum(n: int, target: int, start: int, path: List[int]):
            if len(nums) - start < n or nums[start] * n > target or nums[-1] * n < target: return

            if n == 2:
                l, r = start, len(nums) - 1
                while l < r:
                    s = nums[l] + nums[r]
                    if s == target:
                        ans.append(path + [nums[l], nums[r]])
                        l += 1
                        r -= 1
                        while l < r and nums[l] == nums[l - 1]: l += 1
                        while l < r and nums[r] == nums[r + 1]: r -= 1
                    elif s < target:
                        l += 1
                    else: 
                        r -= 1
                return
            
            for i in range(start, len(nums) - 1):
                if i > start and nums[i - 1] == nums[i]: continue
                nSum(n - 1, target - nums[i], i + 1, path + [nums[i]])

        nSum(4, target, 0, [])
        return ans
```
