# 17. Letter Combinations of a Phone Number

\#Medium #Hash table #String #DFS #BFS

[toc]

## 1. Question

Given a string containing digits from 2-9 inclusive, return all possible letter combinations that the number could represent. Return the answer in **any order**.

A mapping of digits to letters (just like on the telephone buttons) is given below. Note that 1 does not map to any letters.

**Example 1:**

> Input: digits = "23"
> Output: ["ad","ae","af","bd","be","bf","cd","ce","cf"]

**Example 2:**

> Input: digits = "2"
> Output: ["a","b","c"]

**Constraints:**

- $$ 1 \leqq digits.length \leqq 4 $$
- `digits[i]` is a digit in the range `['2', '9']`.

## 2. Solution

### 1) DFS (Recursive)

- Time Complexity: $$ O(n4^n) $$
- Space Complexity: $$ O(n) $$

#### A. C++

- Runtime: 0 ms
- Memory: 9.08MB

```C++
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if (digits.empty()) return {};

        vector<string> ans;

        dfs(digits, 0, "", ans);
        return ans;
    }

private:
    const vector<string> digitToLetters = {"", "", "abc", "def", "ghi", 
        "jkl", "mno", "pqrs", "tuv", "wxyz"};

    //@param path: The combination we are currently building (passed as rvalue reference)
    //By using 'string&&' and 'std::move', Avoids Copies and Reuses the Buffer
    void dfs(const string& digits, int i, string&& path, vector<string>& ans) {
        // the completed 'path' into the result vector
        if (i == digits.length()) {
            ans.push_back(path);
            return;
        };

        for (const char& letter: digitToLetters[digits[i] - '0']) {
            // 1. CHOOSE: Add the letter to our current path
            path.push_back(letter);

            // 2. EXPLORE: Move to the next digit (i + 1)
            // std::move(path) signals that the next level can take ownership of the buffer
            dfs(digits, i + 1, std::move(path), ans);

            // 3. UN-CHOOSE (Backtrack): Remove the letter to try the next one in the loop
            path.pop_back();
        };
    };
    
};
```

#### B. Go

- Runtime: 0 ms
- Memory: 4.05MB

```Go
func letterCombinations(digits string) []string {
    n := len(digits)
    if n == 0 {return nil}

    var ans []string

    digitToLetters := [10]string{"", "", "abc", "def", "ghi",
        "jkl", "mno", "pqrs", "tuv", "wxyz"}

    // Pre-allocate a byte buffer to reuse across recursive calls.
    path := make([]byte, n)

    var dfs func(int)
    dfs = func(i int) {
        if i == n {
            ans = append(ans, string(path))
            return
        }

        letters := digitToLetters[digits[i] - '0']
        for j := 0; j < len(letters); j++ {
            path[i] = letters[j]
            dfs(i + 1)
        }
    }

    dfs(0)
    return ans 
}
```

#### C. Python

- Runtime: 3 ms

```python
class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        n = len(digits)
        if not digits: return []

        ans = []

        digit_to_letters = ['', '', 'abc', 'def', 'ghi', 'jkl',
            'mno', 'pqrs', 'tuv', 'wxyz']

        def dfs(i: int, path: list[str]) -> None:
            if i == n:
                ans.append(''.join(path))
                return
            
            for letter in digit_to_letters[int(digits[i])]:
                path.append(letter)
                dfs(i + 1, path)
                path.pop()
            
        dfs(0, [])
        return ans
```

### 2) BFS (Iterative)

- Time Complexity: $$ O(n4^n) $$
- Space Complexity: $$ O(4^n) $$

#### A. C++

- Runtime: 0 ms
- Memory: 8.92MB

```C++
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if (digits.empty()) return {};

        vector<string> ans{""};
        
        const vector<string> digitToLetters = {"", "", "abc", "def", "ghi", "jkl",
            "mno", "pqrs", "tuv", "wxyz"};

        for (const char& digit: digits){
            vector<string> temp;
            for (const string& s: ans)
                for(const char& c: digitToLetters[digit - '0'])
                    temp.push_back(s + c);
            ans = std::move(temp); //ans takes temp's data, temp becomes empty
        };

        return ans;
    }
};

```

```Go
func letterCombinations(digits string) []string {
    if n == 0 {return nil}

    ans := []string{""}

    digitToLetters := [10]string{"", "", "abc", "def", "ghi",
        "jkl", "mno", "pqrs", "tuv", "wxyz"}

        digit := digits[i] - '0'
        letters := digitToLetters[digit]

        for _, s := range ans {
            for c := 0; c < len(letters); c++ {
                temp = append(temp, s + string(letters[c]))
            }
        }
        
        ans = temp
    }

    return ans
}
```

### 3) IterTools (Cartesian Product)

- Time Complexity: $$ O(n4^n) $$
- Space Complexity: $$ O(4^n) $$

#### Python

- Runtime: 0 ms
- Memory: 19.37MB

```python
class Solution:
    def letterCombinations(self, digits: str) -> list[str]:
        if not digits: return []
        
        digit_to_letters = {
            "2": "abc", "3": "def", "4": "ghi", "5": "jkl",
            "6": "mno", "7": "pqrs", "8": "tuv", "9": "wxyz"
        }
        
        # 1. Get a list of letter groups for each digit: ["abc", "def"]
        letters = [digit_to_letters[digit] for digit in digits]
        
        # 2. Use product() to find all combinations: ('a', 'd'), ('a', 'e')...
        # 3. Join each tuple into a string
        return ["".join(combo) for combo in product(*letters)]
```
